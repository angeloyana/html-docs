    Unordered Lists

    <ul type="disc"> <!-- (1)! -->
  <li>Item 1</li>
  <li>Item 2</li>
</ul>
<ul type="square">
  <li>Item 1</li>
  <li>Item 2</li>
</ul>
<ul type="circle">
  <li>Item 1</li>
  <li>Item 2</li>
</ul>
